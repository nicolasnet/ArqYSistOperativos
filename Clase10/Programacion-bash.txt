	NC='\033[0m'    (para restaurar el color)

		echo "usuario Correcto: root"
	elif [ ${USER} == 'juan']; then
		echo "usuario Corerecto: juan"
	else
		echo -e "Usuario Incorrecto " ${ROJO}${USER}${NC}
		exit 1
